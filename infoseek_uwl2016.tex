\frame{
  \ft{Literature}

  \uncover<+->{
  \begin{block}{\textit{Convenience} is a driving factor for search engine choice}
    \bi
    \item<+-> Popular search engines are most popular
    \item<+-> To blame: Volume and complexity of library search tools (Lee, H., 2008; Connaway et al., 2011; Taylor, 2012; Lee, J., 2012; Rempel, 2013)
    \item<+-> Georgas (2013) directly observed students... not knowing what they were looking at...
    \item<+-> Still, students have a high self evaluation of their research skills (Georgas 2013, 2014)
    \ei      
  \end{block}}

  \uncover<+->{
  \begin{block}{No change with academic progression}
    \bi
    \item Mbabu et al. (2012): Juniors and seniors used the subscription databases less than freshmen and sophomores.
    \item Callinan (2005): Both first-year and final-year students struggled to use library electronic resources.
      \ei
  \end{block}}
}

\frame{
  \ft{Factors that influence information seeking behavior}

  \footnotesize{
  \begin{block}{Variables Staff / Instructors Control}
    \bi
    \item \textit{Encourage:} Do instructors encourage students to use electronic databases?  (=1 if frequently)
    \item \textit{List:} Do instructors give reading lists? (=1 if frequently/occasionally)
    \item \textit{ClassVisit:} Did a library staff member visit their class? (=1 if so)
      \ei
  \end{block}


  \begin{block}{Academic Variables}
    \bi
    \item \textit{ColBus} and \textit{ColSci}: Binary variable for colleges of business and science, respectively.  Omitted category is College of Liberal Studies.
    \item \textit{ACT}: College entrance exam score
      \ei
  \end{block}

  \begin{block}{Demographic Variables}
    \bi
    \item \textit{Age}
    \item \textit{ParentEdu:} =1 if at least one parent has 4-year college degree 
    \item \textit{Female:} =1 if student identifies as female
    \item \textit{Non-white:} =1 if student identifies race as something other than white
    \ei
  \end{block}
  }
    
}

\frame {
  \ft{Conclusions}
  \uncover<+->{
  \begin{block}{Good and questionable information-seeking behavior}
    \bi
    \item We have a lot of students using library electronic databases and finding scholarly articles
    \item We also have a lot of students using Google.com and Wikipedia.org
    \item Students have presence on campus and in library
    \ei
  \end{block}
  }

  \uncover<+->{
  \begin{block}{Staff and instructors can make a difference}
    \bi
    \item Instructors encouragement leads to more use of electronic databases and scholarly articles
    \item Giving reading lists leads to more value for peer-review, but less use of electronic databases
    \item Class visits by library staff leads to higher use of electronic databases, higher value for peer-review
    \item More need in colleges of business and science/health
    \ei
  \end{block}
  }
